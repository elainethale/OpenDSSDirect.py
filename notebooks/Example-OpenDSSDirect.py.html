b''

To get all kW, one must iterate over all the loads

In [10]:

dss.Loads.First()

while True:

    print(
        'Name={name} \t kW={kW}'.format(
            name=dss.Loads.Name(),
            kW=dss.Loads.kW()
        )
    )

    if not dss.Loads.Next() > 0:
        break
